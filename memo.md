	kz = y - p
z = l * x + p
	lx = z - p 

lx = lhy + lp
z - p = lhy + lp
z = lhy + lp + p
kz = lhky + lkp + kp

y = lhky + lkp + kp + p
y - lhky = p(lk + k + 1)
y(1 - lhk) = p(lk + k + 1)
p = y(1 - lhk) / (lk + k + 1)
